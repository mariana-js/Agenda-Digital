CREATE TABLE SETOR (
	ID_SETOR INTEGER NOT NULL UNIQUE CONSTRAINT PK_ID_SETOR PRIMARY KEY,
	NOME_SETOR VARCHAR(20)
	SIGLA_SETOR VARCHAR(5)

CREATE TABLE RAMAL (
	NUMERO_RAMAL VARCHAR(10) UNIQUE CONSTRAINT PK_NUMERO_RAMAL PRIMARY KEY
);

CREATE TABLE SETOR_RAMAL (
	ID_RAMAL_SETOR INTEGER NOT NULL UNIQUE CONSTRAINT PK_ID_RAMAL PRIMARY KEY,
	ID_SETOR_RAMAL INTEGER NOT NULL UNIQUE CONSTRAINT PK_ID_SETOR PRIMARY KEY,
	NUMERO_RAMAL VARCHAR(10)
);

CREATE TABLE ENDERECO (
	ID_ENDERECO INTEGER NOT NULL UNIQUE CONSTRAINT PK_ID_ENDERECO PRIMARY KEY,
	ESTADO VARCHAR(20),
	LOGADOURO VARCHAR(40),
	CIDADE VARCHAR(20),
	CEP VARCHAR(12)
);

CREATE TABLE PESSOA (
	ID_PESSOA INTEGER NOT NULL UNIQUE CONSTRAINT PK_ID_PESSOA PRIMARY KEY,
	NOME_PESSOA VARCHAR(50),
	EMAIL VARCHAR(50) UNIQUE,
	CELULAR_CORPORATIVO VARCHAR(16),
	CELULAR_PESSOAL VARCHAR(16) UNIQUE,
	TELEFONE VARCHAR(16),
	FLAG_PRIVADO BOOLEAN ,
	FLAG_FUNCIONARIO BOOLEAN
);

CREATE TABLE FUNCIONARIO (
	ID_FUNCIONARIO INTEGER NOT NULL UNIQUE CONSTRAINT PK_ID_FUNCIONARIO PRIMARY KEY,
	IMG BYTEA,
	DATA_NACIMENTO DATE NOT NULL,
	FLAG_ADMINISTRADOR BOOLEAN
) INHERITS(PESSOA);
